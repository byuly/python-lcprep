Build from scratch in interviews. Use dict of dicts (Pythonic) or TrieNode class. Used for: autocomplete, word search, prefix matching.
Dict-of-dicts (concise)

END = "#"

    for c in word:
        node = node[c]

TrieNode class
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word):
        n = self.root
        for c in word:
            if c not in n.children:
                n.children[c] = TrieNode()
            n = n.children[c]
        n.is_end = True

    def search(self, word):
        n = self._find(word)
        return n is not None and n.is_end

    def _find(self, prefix):
        n = self.root
        for c in prefix:
            if c not in n.children:
                return None
            n = n.children[c]
        return n
Linked list operations
O(n)▼
Dummy head simplifies edge cases. Most linked list problems use two-pointer or prev/curr/next manipulation. Always handle head = None.
Reverse (iterative)
def reverseList(head):
    prev, curr = None, head
    while curr:
        nxt = curr.next
        curr.next = prev
        prev = curr
        curr = nxt
    return prev
Merge two sorted lists
def mergeTwoLists(l1, l2):
    dummy = curr = ListNode(0)
    while l1 and l2:
        if l1.val <= l2.val:
            curr.next = l1; l1 = l1.next
        else:
            curr.next = l2; l2 = l2.next
        curr = curr.next
    curr.next = l1 or l2
    return dummy.next

def removeNthFromEnd(head, n):

        fast = fast.next
        slow = slow.next

Dijkstra's shortest path
O((V+E) log V)▼
Use for weighted graphs with non-negative edges. Use BFS for unweighted. The lazy-deletion version below is simplest to write in interviews.
import heapq
from collections import defaultdict

def dijkstra(graph, src, n):
    # graph[u] = [(weight, v), ...]
    dist = [float('inf')] * n
    dist[src] = 0
    heap = [(0, src)]        # (dist, node)

    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]: continue  # lazy delete
        for w, v in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(heap, (dist[v], v))
    return dist
Union Find
O(α(n)) ≈ O(1)▼
Always include path compression + union by rank. Used for: connected components, cycle detection, minimum spanning tree.
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
        self.components = n

    def find(self, x):            # path compression
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, a, b):       # union by rank
        pa, pb = self.find(a), self.find(b)
        if pa == pb: return False
        if self.rank[pa] < self.rank[pb]:
            pa, pb = pb, pa
        self.parent[pb] = pa
        if self.rank[pa] == self.rank[pb]:
            self.rank[pa] += 1
        self.components -= 1
        return True

    def connected(self, a, b):
        return self.find(a) == self.find(b)

O(n!) worst▼
Pattern: choose → explore → unchoose. In Python, path[:] or list(path) makes a copy. Sort input first for deduplication.
res = []
def bt(start, path):

Interval patterns
O(n log n)▼
Always sort by start time first. Then scan left→right, merging or comparing with the previous interval. Meeting rooms II uses a min-heap for concurrent rooms.
Merge overlapping intervals
def merge(intervals):
    intervals.sort(key=lambda x: x[0])
    res = [intervals[0]]
    for s, e in intervals[1:]:
        if s <= res[-1][1]:   # overlap
            res[-1][1] = max(res[-1][1], e)
        else:
            res.append([s, e])
    return res
Insert interval
def insert(intervals, new):
    res, i = [], 0
    # Add all before new
    while i < len(intervals) \
          and intervals[i][1] < new[0]:
        res.append(intervals[i]); i += 1
    # Merge overlapping with new
    while i < len(intervals) \
          and intervals[i][0] <= new[1]:
        new = [min(new[0], intervals[i][0]),
               max(new[1], intervals[i][1])]
        i += 1
    res.append(new)
    return res + intervals[i:]
Meeting rooms II (min-heap for end times)
import heapq

def minMeetingRooms(intervals):
    intervals.sort(key=lambda x: x[0])
    heap = []   # end times of active meetings
    for s, e in intervals:
        if heap and heap[0] <= s:
            heapq.heappop(heap)   # reuse room
        heapq.heappush(heap, e)
    return len(heap)             # = max concurrent
Greedy algorithms
varies▼
Make the locally optimal choice at each step. Greedy works when local optimum leads to global optimum. Key insight: prove that skipping a greedy choice never helps.
Jump Game I & II
# I — can you reach the end?
def canJump(nums):
    reach = 0
    for i, n in enumerate(nums):
        if i > reach: return False
        reach = max(reach, i + n)
    return True

# II — min jumps to reach end
def jump(nums):
    jumps = end = farthest = 0
    for i in range(len(nums) - 1):
        farthest = max(farthest, i + nums[i])
        if i == end:
            jumps += 1; end = farthest
    return jumps
Gas station (circular)
def canCompleteCircuit(gas, cost):
    if sum(gas) < sum(cost):
        return -1
    start = tank = 0
    for i in range(len(gas)):
        tank += gas[i] - cost[i]
        if tank < 0:
            start = i + 1
            tank = 0
    return start
Task scheduler
def leastInterval(tasks, n):
    freq = Counter(tasks)
    max_f = max(freq.values())
    max_count = sum(
        1 for f in freq.values()
        if f == max_f)
    return max(len(tasks),
        (max_f - 1) * (n + 1) + max_count)
Partition labels
def partitionLabels(s):
    last = {c: i for i, c in enumerate(s)}
    res, start, end = [], 0, 0
    for i, c in enumerate(s):
        end = max(end, last[c])
        if i == end:
            res.append(i - start + 1)
            start = i + 1
    return res

    top, bot = 0, len(matrix) - 1
    left, right = 0, len(matrix[0]) - 1
    while top <= bot and left <= right:
        for c in range(left, right+1):
            res.append(matrix[top][c])
        top += 1
        for r in range(top, bot+1):
            res.append(matrix[r][right])
        right -= 1
        if top <= bot:
            for c in range(right, left-1, -1):
                res.append(matrix[bot][c])
            bot -= 1
        if left <= right:
            for r in range(bot, top-1, -1):
                res.append(matrix[r][left])
            left += 1
    return res

# Rotate 90° clockwise: transpose + reverse
def rotate(matrix):
    n = len(matrix)
    for i in range(n):
        for j in range(i+1, n):
            matrix[i][j], matrix[j][i] = \
                matrix[j][i], matrix[i][j]
    for row in matrix:
        row.reverse()

# Search sorted 2D (staircase)
def searchMatrix(matrix, target):
    r, c = 0, len(matrix[0]) - 1
    while r < len(matrix) and c >= 0:
        if matrix[r][c] == target:
            return True
        elif matrix[r][c] > target:
            c -= 1
        else:
            r += 1
    return False

Python pitfalls & traps
watch out▼
⚠Mutable default argument: def f(lst=[]) shares the same list across all calls. Use def f(lst=None): lst = lst or [].
⚠Shallow copy trap: b = a copies a reference. b = a[:] or b = list(a) copies the list. For 2D: use copy.deepcopy() or [row[:] for row in grid].
⚠List multiplication 2D: [[0]*n]*m creates m references to the same row. Always use [[0]*n for _ in range(m)].
⚠Modifying list while iterating: results in skipped or duplicated elements. Iterate over a copy: for x in lst[:]: or collect and modify after.
⚠list.pop(0) is O(n). Use collections.deque with popleft() for O(1) front removal.
⚠Backtracking: res.append(path) appends a reference. Always append a copy: res.append(path[:]).
⚠Walrus operator scope: if (n := len(a)) > 10: — n leaks into the enclosing scope. Useful for while-loops: while chunk := f.read(8192):.
⚠Integer division: Python's // floors toward negative infinity (-7//2 == -4), not toward zero like Java/C (-7/2 == -3).
⚠Global/nonlocal: In nested functions, you must declare nonlocal x to reassign an outer variable. Reading is fine without it.
⚠heapq is min-heap only. To simulate max-heap, push/pop negated values. For tuples, Python compares lexicographically, so (priority, idx, item) breaks ties.

Recursion anatomy & mental model
foundation▼
Every recursive function needs: (1) a base case that stops, (2) a recursive call that moves toward it, (3) trust — assume the recursive call returns the correct answer and build on top of it.
Anatomy of a recursive function
def solve(problem):
    # 1. BASE CASE — stop condition (often None, 0, empty)
    if problem is small_enough:
        return base_answer

    # 2. REDUCE — make the problem smaller
    smaller = reduce(problem)

    # 3. RECURSE — trust it returns correct answer
    sub_answer = solve(smaller)

    # 4. COMBINE — build your answer from sub_answer
    return combine(sub_answer, problem)
Classic examples
# Factorial
def factorial(n):
    if n <= 1: return 1        # base
    return n * factorial(n-1)  # reduce + combine

# Fibonacci (naive — use @cache in practice)
def fib(n):
    if n <= 1: return n
    return fib(n-1) + fib(n-2)

# Power
def power(x, n):
    if n == 0: return 1
    if n % 2 == 0:             # fast power O(log n)
        half = power(x, n//2)
        return half * half
    return x * power(x, n-1)

LeetCode wraps solutions in a Solution class. You call helper methods with self.helper() and store shared state as instance variables. This is the most common interview pattern.
Standard LeetCode class pattern

    def maxDepth(self, root) -> int:
        # self is the Solution instance — gives you access
        # to other methods you define in the class
        if not root: return 0
        return 1 + max(self.maxDepth(root.left),
                        self.maxDepth(root.right))
        # ^ calls itself via self — equivalent to maxDepth(root.left)
        #   but required when inside a class
self as shared state (accumulator)
class Solution:
    def diameterOfBinaryTree(self, root) -> int:
        self.ans = 0    # instance var — shared across all
                        # recursive calls, no nonlocal needed
        def depth(node):
            if not node: return 0
            L = depth(node.left)
            R = depth(node.right)
            self.ans = max(self.ans, L + R)  # update via self
            return 1 + max(L, R)
        depth(root)
        return self.ans
self.helper() — separate method
class Solution:
    def pathSum(self, root, target) -> int:
        self.count = 0
        def dfs(node, curr):
            if not node: return
            curr += node.val
            if curr == target:
                self.count += 1
            dfs(node.left, curr)
            dfs(node.right, curr)
        dfs(root, 0)
        return self.count

    # Alternatively, delegate to a helper method:
    def _dfs(self, node, curr, target):
        if not node: return 0
        curr += node.val
        same = 1 if curr == target else 0
        return same + self._dfs(node.left, curr, target) \
                     + self._dfs(node.right, curr, target)

Tree recursion — return multiple values
O(n)▼
The most powerful tree pattern: return a tuple of everything you need from a subtree so the parent can compute its answer bottom-up. Avoids global state entirely.
Balanced binary tree
class Solution:
    def isBalanced(self, root) -> bool:
        def check(node):
            # returns (is_balanced, height)
            if not node: return True, 0
            lb, lh = check(node.left)
            rb, rh = check(node.right)
            balanced = lb and rb and abs(lh-rh) <= 1
            return balanced, 1 + max(lh, rh)

Camera surveillance (3 states)
class Solution:
    def minCameraCover(self, root) -> int:
        self.cams = 0
        def dfs(node):
            # returns: 0=uncovered, 1=covered, 2=has_camera
            if not node: return 1   # null = covered
            l, r = dfs(node.left), dfs(node.right)
            if l == 0 or r == 0:    # child needs coverage
                self.cams += 1; return 2
            if l == 2 or r == 2:    # child has camera
                return 1
            return 0               # need parent to cover
        if dfs(root) == 0: self.cams += 1
        return self.cams
Lowest common ancestor (LCA)
class Solution:
    def lowestCommonAncestor(self, root, p, q):
        def dfs(node):
            if not node or node == p or node == q:
                return node           # found one or null
            left  = dfs(node.left)
            right = dfs(node.right)
            # if both sides returned something, this is LCA
            if left and right: return node
            return left or right      # propagate up
        return dfs(root)
Memoized recursion — @cache / @lru_cache
top-down DP▼
@cache is the single most powerful tool for recursive DP in Python. It turns exponential recursion into polynomial by caching every (args) → result mapping automatically.
@cache — simplest top-down DP
from functools import cache

# Coin change — O(amount * len(coins))
class Solution:
    def coinChange(self, coins, amount) -> int:
        @cache              # decorates inner function
        def dp(rem):
            if rem == 0: return 0
            if rem < 0: return float('inf')
            return 1 + min(dp(rem - c) for c in coins)
        res = dp(amount)
        dp.cache_clear()    # IMPORTANT: clear between test cases!
        return res if res != float('inf') else -1
@lru_cache on class method
from functools import lru_cache

class Solution:
    def longestPalin(self, s: str) -> int:
        n = len(s)
        # lru_cache on nested function captures s
        @lru_cache(maxsize=None)
        def dp(i, j):
            if i > j: return 0
            if i == j: return 1
            if s[i] == s[j]:
                return 2 + dp(i+1, j-1)
            return max(dp(i+1, j), dp(i, j-1))
        return dp(0, n-1)
Explicit memo dict (most flexible)
class Solution:
    def uniquePaths(self, m, n) -> int:
        memo = {}
        def dp(r, c):
            if r == 0 or c == 0: return 1
            if (r,c) in memo: return memo[(r,c)]
            res = dp(r-1, c) + dp(r, c-1)
            memo[(r,c)] = res
            return res
        return dp(m-1, n-1)
@cache gotcha When @cache/@lru_cache decorates a nested function, it holds a reference to the enclosing scope (like coins or s). Always call dp.cache_clear() after use, or the cache persists across LeetCode test cases and causes Wrong Answer on later tests.
Divide & conquer recursion
O(n log n)▼
Split problem in half, recurse on each half, combine results. The key is identifying the combine step — that's usually where the work happens.
Merge sort (implement from scratch)
def merge_sort(arr):
    if len(arr) <= 1: return arr  # base case
    mid = len(arr) // 2
    L = merge_sort(arr[:mid])
    R = merge_sort(arr[mid:])
    return merge(L, R)

def merge(L, R):
    res, i, j = [], 0, 0
    while i < len(L) and j < len(R):
        if L[i] <= R[j]: res.append(L[i]); i += 1
        else:            res.append(R[j]); j += 1
    return res + L[i:] + R[j:]
Count inversions (D&C)
def count_inversions(arr):
    if len(arr) <= 1: return arr, 0
    mid = len(arr) // 2
    L, lc = count_inversions(arr[:mid])
    R, rc = count_inversions(arr[mid:])
    merged, mc = merge_count(L, R)
    return merged, lc + rc + mc

def merge_count(L, R):
    res, inv, i, j = [], 0, 0, 0
    while i < len(L) and j < len(R):
        if L[i] <= R[j]: res.append(L[i]); i += 1
        else:
            inv += len(L) - i  # all remaining L > R[j]
            res.append(R[j]); j += 1
    return res + L[i:] + R[j:], inv

def is_palindrome(s, l=0, r=None):

# Reverse a linked list recursively
def reverseList(head):
    if not head or not head.next:
        return head                 # base: 1 node
    new_head = reverseList(head.next)
    head.next.next = head           # reverse pointer
    head.next = None
    return new_head

Tail recursion, stack depth & converting to iterative
Python limits▼
Python does NOT optimize tail calls. Every recursive call adds a frame regardless. For deep inputs (n > 1000), either increase the limit or rewrite iteratively. Know how to convert any recursion to a stack-based loop.
Tail recursion (Python ignores it, but good form)
# Non-tail: builds up call stack waiting for results
def factorial(n):
    if n == 1: return 1
    return n * factorial(n-1)  # waits for result

# Tail-recursive form (accumulator pattern)
def factorial(n, acc=1):
    if n == 1: return acc
    return factorial(n-1, n*acc)  # last op = the call
# Python still creates n frames for both — no TCO!

# True fix: convert to iterative
def factorial(n):
    acc = 1
    while n > 1: acc *= n; n -= 1
    return acc
Convert any DFS to iterative
# Recursive preorder
def preorder(node):
    if not node: return
    visit(node)
    preorder(node.left)
    preorder(node.right)

# Iterative equivalent — simulate call stack
def preorder_iter(root):
    if not root: return
    stack = [root]
    while stack:
        node = stack.pop()
        visit(node)
        # right first → left processed first (LIFO)
        if node.right: stack.append(node.right)
        if node.left:  stack.append(node.left)
General recursion → iteration template
# Any recursive function can become iterative
# by pushing (node, state) tuples onto a stack

# Postorder iterative (tricky — two visits per node)
def postorder_iter(root):
    stack, res = [(root, False)], []
    while stack:
        node, visited = stack.pop()
        if not node: continue
        if visited:
            res.append(node.val)  # second visit → process
        else:
            # Push self (to revisit) then children
            stack.append((node, True))
            stack.append((node.right, False))
            stack.append((node.left, False))
    return res
Recursion decision tree — all major patterns
reference▼
How to structure any recursive solution Step 1 — Identify the subproblem. What is one recursive call computing? "Returns the max path sum from this node downward." Be precise.

Step 2 — Write the base case first. Empty/null/zero/index-out-of-bounds. Get it wrong and you get infinite recursion.

Step 3 — Assume correctness, combine. If dp(i) returns the right answer for smaller inputs, how do you use it for dp(i-1), dp(i-2) etc.?

Step 4 — Decide the return vs side effect trade-off. Return values = clean and composable. Side effects via self.x or nonlocal = simpler when you just need a running total.
Pattern map — pick the right skeleton
# 1. LINEAR — one recursive call, reducing index/node
