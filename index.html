class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

class Trie:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word):
        n = self.root
        for c in word:
            if c not in n.children:
                n.children[c] = TrieNode()
            n = n.children[c]
        n.is_end = True

    def search(self, word):
        n = self._find(word)
        return n is not None and n.is_end

    def _find(self, prefix):
        n = self.root
        for c in prefix:
            if c not in n.children:
                return None
            n = n.children[c]
        return n

Dijkstra's shortest path
O((V+E) log V)▼
Use for weighted graphs with non-negative edges. Use BFS for unweighted. The lazy-deletion version below is simplest to write in interviews.
import heapq
from collections import defaultdict

def dijkstra(graph, src, n):
    # graph[u] = [(weight, v), ...]
    dist = [float('inf')] * n
    dist[src] = 0
    heap = [(0, src)]        # (dist, node)

    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]: continue  # lazy delete
        for w, v in graph[u]:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                heapq.heappush(heap, (dist[v], v))
    return dist
Union Find
O(α(n)) ≈ O(1)▼
Always include path compression + union by rank. Used for: connected components, cycle detection, minimum spanning tree.
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
        self.components = n

    def find(self, x):            # path compression
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, a, b):       # union by rank
        pa, pb = self.find(a), self.find(b)
        if pa == pb: return False
        if self.rank[pa] < self.rank[pb]:
            pa, pb = pb, pa
        self.parent[pb] = pa
        if self.rank[pa] == self.rank[pb]:
            self.rank[pa] += 1
        self.components -= 1
        return True

    def connected(self, a, b):
        return self.find(a) == self.find(b)

Interval patterns
O(n log n)▼
Always sort by start time first. Then scan left→right, merging or comparing with the previous interval. Meeting rooms II uses a min-heap for concurrent rooms.
Merge overlapping intervals
def merge(intervals):
    intervals.sort(key=lambda x: x[0])
    res = [intervals[0]]
    for s, e in intervals[1:]:
        if s <= res[-1][1]:   # overlap
            res[-1][1] = max(res[-1][1], e)
        else:
            res.append([s, e])
    return res
Insert interval
def insert(intervals, new):
    res, i = [], 0
    # Add all before new
    while i < len(intervals) \
          and intervals[i][1] < new[0]:
        res.append(intervals[i]); i += 1
    # Merge overlapping with new
    while i < len(intervals) \
          and intervals[i][0] <= new[1]:
        new = [min(new[0], intervals[i][0]),
               max(new[1], intervals[i][1])]
        i += 1
    res.append(new)
    return res + intervals[i:]

import heapq

def minMeetingRooms(intervals):
    intervals.sort(key=lambda x: x[0])
    heap = []   # end times of active meetings
    for s, e in intervals:
        if heap and heap[0] <= s:
            heapq.heappop(heap)   # reuse room
        heapq.heappush(heap, e)
    return len(heap)             # = max concurrent

Make the locally optimal choice at each step. Greedy works when local optimum leads to global optimum. Key insight: prove that skipping a greedy choice never helps.
Jump Game I & II
# I — can you reach the end?
def canJump(nums):
    reach = 0
    for i, n in enumerate(nums):
        if i > reach: return False
        reach = max(reach, i + n)
    return True

# II — min jumps to reach end
def jump(nums):
    jumps = end = farthest = 0
    for i in range(len(nums) - 1):
        farthest = max(farthest, i + nums[i])
        if i == end:
            jumps += 1; end = farthest
    return jumps
Gas station (circular)
def canCompleteCircuit(gas, cost):
    if sum(gas) < sum(cost):
        return -1
    start = tank = 0
    for i in range(len(gas)):
        tank += gas[i] - cost[i]
        if tank < 0:
            start = i + 1
            tank = 0
    return start
Task scheduler
def leastInterval(tasks, n):
    freq = Counter(tasks)
    max_f = max(freq.values())
    max_count = sum(
        1 for f in freq.values()
        if f == max_f)
    return max(len(tasks),
        (max_f - 1) * (n + 1) + max_count)

    left, right = 0, len(matrix[0]) - 1
    while top <= bot and left <= right:
        for c in range(left, right+1):
            res.append(matrix[top][c])
        top += 1
        for r in range(top, bot+1):
            res.append(matrix[r][right])
        right -= 1
        if top <= bot:
            for c in range(right, left-1, -1):
                res.append(matrix[bot][c])
            bot -= 1
        if left <= right:
            for r in range(bot, top-1, -1):
                res.append(matrix[r][left])
            left += 1
    return res

Python pitfalls & traps
watch out▼
⚠Mutable default argument: def f(lst=[]) shares the same list across all calls. Use def f(lst=None): lst = lst or [].
⚠Shallow copy trap: b = a copies a reference. b = a[:] or b = list(a) copies the list. For 2D: use copy.deepcopy() or [row[:] for row in grid].
⚠List multiplication 2D: [[0]*n]*m creates m references to the same row. Always use [[0]*n for _ in range(m)].
⚠Modifying list while iterating: results in skipped or duplicated elements. Iterate over a copy: for x in lst[:]: or collect and modify after.
⚠list.pop(0) is O(n). Use collections.deque with popleft() for O(1) front removal.
⚠Backtracking: res.append(path) appends a reference. Always append a copy: res.append(path[:]).
⚠Walrus operator scope: if (n := len(a)) > 10: — n leaks into the enclosing scope. Useful for while-loops: while chunk := f.read(8192):.
⚠Integer division: Python's // floors toward negative infinity (-7//2 == -4), not toward zero like Java/C (-7/2 == -3).
⚠Global/nonlocal: In nested functions, you must declare nonlocal x to reassign an outer variable. Reading is fine without it.
⚠heapq is min-heap only. To simulate max-heap, push/pop negated values. For tuples, Python compares lexicographically, so (priority, idx, item) breaks ties.

Recursion anatomy & mental model
foundation▼
Every recursive function needs: (1) a base case that stops, (2) a recursive call that moves toward it, (3) trust — assume the recursive call returns the correct answer and build on top of it.
Anatomy of a recursive function
def solve(problem):
    # 1. BASE CASE — stop condition (often None, 0, empty)
    if problem is small_enough:
        return base_answer

    # 2. REDUCE — make the problem smaller
    smaller = reduce(problem)

    # 3. RECURSE — trust it returns correct answer
    sub_answer = solve(smaller)

    # 4. COMBINE — build your answer from sub_answer
    return combine(sub_answer, problem)

    if n % 2 == 0:             # fast power O(log n)

Memoized recursion — @cache / @lru_cache
top-down DP▼
@cache is the single most powerful tool for recursive DP in Python. It turns exponential recursion into polynomial by caching every (args) → result mapping automatically.
@cache — simplest top-down DP
from functools import cache

# Coin change — O(amount * len(coins))
class Solution:
    def coinChange(self, coins, amount) -> int:
        @cache              # decorates inner function
        def dp(rem):
            if rem == 0: return 0
            if rem < 0: return float('inf')
            return 1 + min(dp(rem - c) for c in coins)
        res = dp(amount)
        dp.cache_clear()    # IMPORTANT: clear between test cases!
        return res if res != float('inf') else -1
@lru_cache on class method
from functools import lru_cache

class Solution:
    def longestPalin(self, s: str) -> int:
        n = len(s)
        # lru_cache on nested function captures s
        @lru_cache(maxsize=None)
        def dp(i, j):
            if i > j: return 0
            if i == j: return 1
            if s[i] == s[j]:
                return 2 + dp(i+1, j-1)
            return max(dp(i+1, j), dp(i, j-1))
        return dp(0, n-1)
Explicit memo dict (most flexible)
class Solution:
    def uniquePaths(self, m, n) -> int:
        memo = {}
        def dp(r, c):
            if r == 0 or c == 0: return 1
            if (r,c) in memo: return memo[(r,c)]
            res = dp(r-1, c) + dp(r, c-1)
            memo[(r,c)] = res
            return res
        return dp(m-1, n-1)

def merge_sort(arr):
    if len(arr) <= 1: return arr  # base case
    mid = len(arr) // 2
    L = merge_sort(arr[:mid])
    R = merge_sort(arr[mid:])
    return merge(L, R)

def merge(L, R):
    res, i, j = [], 0, 0
    while i < len(L) and j < len(R):
        if L[i] <= R[j]: res.append(L[i]); i += 1
        else:            res.append(R[j]); j += 1
    return res + L[i:] + R[j:]

    while i < len(L) and j < len(R):

Tail recursion, stack depth & converting to iterative
Python limits▼
Python does NOT optimize tail calls. Every recursive call adds a frame regardless. For deep inputs (n > 1000), either increase the limit or rewrite iteratively. Know how to convert any recursion to a stack-based loop.
Tail recursion (Python ignores it, but good form)
# Non-tail: builds up call stack waiting for results
def factorial(n):
    if n == 1: return 1
    return n * factorial(n-1)  # waits for result

# Tail-recursive form (accumulator pattern)
def factorial(n, acc=1):
    if n == 1: return acc
    return factorial(n-1, n*acc)  # last op = the call
# Python still creates n frames for both — no TCO!

# True fix: convert to iterative
def factorial(n):
    acc = 1
    while n > 1: acc *= n; n -= 1
    return acc
Convert any DFS to iterative
# Recursive preorder
def preorder(node):
    if not node: return
    visit(node)
    preorder(node.left)
    preorder(node.right)

# Iterative equivalent — simulate call stack
def preorder_iter(root):
    if not root: return
    stack = [root]
    while stack:
        node = stack.pop()
        visit(node)
        # right first → left processed first (LIFO)
        if node.right: stack.append(node.right)
        if node.left:  stack.append(node.left)
General recursion → iteration template
# Any recursive function can become iterative
# by pushing (node, state) tuples onto a stack

# Postorder iterative (tricky — two visits per node)
def postorder_iter(root):
    stack, res = [(root, False)], []
    while stack:
        node, visited = stack.pop()
        if not node: continue
        if visited:
            res.append(node.val)  # second visit → process
        else:
            # Push self (to revisit) then children
            stack.append((node, True))
            stack.append((node.right, False))
            stack.append((node.left, False))
    return res
Recursion decision tree — all major patterns
reference▼
How to structure any recursive solution Step 1 — Identify the subproblem. What is one recursive call computing? "Returns the max path sum from this node downward." Be precise.

Step 2 — Write the base case first. Empty/null/zero/index-out-of-bounds. Get it wrong and you get infinite recursion.

Step 3 — Assume correctness, combine. If dp(i) returns the right answer for smaller inputs, how do you use it for dp(i-1), dp(i-2) etc.?

Step 4 — Decide the return vs side effect trade-off. Return values = clean and composable. Side effects via self.x or nonlocal = simpler when you just need a running total.
Pattern map — pick the right skeleton

def f(i): return base if i==end else work + f(i+1)

# 2. BINARY TREE — two recursive calls
def f(node): return base if not node else combine(f(node.left), f(node.right))

# 3. INCLUDE / EXCLUDE — two choices at each step
def f(i): return max(skip(f(i+1)), take(nums[i]+f(i+2)))

# 4. BACKTRACKING — choose → explore → unchoose
def bt(i, path):
    if done: results.append(path[:]); return
    for c in choices(i):
        path.append(c); bt(i+1, path); path.pop()

# 5. DIVIDE & CONQUER — split in half, merge
def f(arr):
    if small: return base
    mid = len(arr)//2
    return merge(f(arr[:mid]), f(arr[mid:]))

# 6. MEMOIZED (top-down DP) — linear or 2D
@cache
def f(i, j): return base if edge else recurrence(f(i-1,j), f(i,j-1))

Dunder methods & operator overloading
protocol▼
Dunder methods are how custom objects plug into built-in syntax (+, len(), in, for) and functions (sorted(), heapq). __eq__ and __hash__ must agree — equal objects must hash equal, or the object silently breaks in sets/dicts.
Core dunders for interview-grade classes
class Point:
    def __init__(self, x, y):
        self.x, self.y = x, y

    def __repr__(self):              # debug repr — unambiguous
        return f"Point({self.x}, {self.y})"

    def __eq__(self, other):
        return (self.x, self.y) == (other.x, other.y)

    def __hash__(self):              # required if you define __eq__
        return hash((self.x, self.y))  # and want it in a set/dict key

    def __lt__(self, other):         # enables sorted() / heapq
        return (self.x, self.y) < (other.x, other.y)

    def __add__(self, other):        # operator overloading: p1 + p2
        return Point(self.x + other.x, self.y + other.y)
Container & callable protocols
class Vector:

        self.data = data

    def __len__(self):             # enables len(v)
        return len(self.data)

    def __getitem__(self, i):        # enables v[i], slicing, iteration
        return self.data[i]

        return x in self.data

        return iter(self.data)

Under the hood — lists & strings
CPython internals▼
Lists are arrays of pointers (PyObject*), not raw values — even [1, 2, 3] stores 3 pointers to separately-allocated int objects. Strings are immutable arrays of Unicode code points — every "mutation" allocates a brand-new string.
List: dynamic array with over-allocation

# space -> O(1); occasional growth reallocates + copies -> O(n).

# => O(1) average, O(n) worst case (bad hash / many collisions)

Balanced trees — AVL & Red-Black
follow-up favorite▼
Python has no built-in self-balancing BST — dict/set are hash tables, not ordered trees. For ordered operations (range queries, floor/ceil, k-th element) with guaranteed O(log n), reach for sortedcontainers.SortedList in practice — but be able to explain the theory from scratch.
AVL tree — balance by height
# Invariant: for every node, |height(left) - height(right)| <= 1
# Rebalance after insert/delete via rotations, checked bottom-up

class AVLNode:
    def __init__(self, val):
        self.val, self.left, self.right, self.height = val, None, None, 1

def height(n): return n.height if n else 0
def balance_factor(n): return height(n.left) - height(n.right)

def rotate_right(y):           # y is unbalanced, left-heavy
    x, t = y.left, y.left.right
    x.right, y.left = y, t
    y.height = 1 + max(height(y.left), height(y.right))
    x.height = 1 + max(height(x.left), height(x.right))
    return x                   # x is the new subtree root
Red-Black tree — balance by color rules
# Invariants (looser than AVL -> cheaper writes, still O(log n)):
# 1. Every node is red or black
# 2. Root is black
# 3. A red node's children are both black (no two reds in a row)
# 4. Every root->None path has the same number of black nodes

# Rebalance after insert = recolor + up to 2 rotations, vs AVL's
# possibly-many rotations up the path. This is WHY std::map (C++)
# and TreeMap (Java) use red-black trees: cheaper writes, at the
# cost of a slightly less-tight balance (taller by a constant factor)

# AVL:       stricter balance -> faster lookups, slower writes
# Red-Black: looser balance   -> faster writes, slightly slower lookups

Iterators, generators & the iteration protocol
protocol▼
for x in obj desugars to iter(obj) then repeated next() calls until StopIteration. A generator function (uses yield) builds an object that implements this protocol for you, pausing/resuming its own frame between values instead of computing everything up front.
The protocol, written by hand
class CountUp:
    def __init__(self, n): self.n = n
    def __iter__(self):          # iterable -> returns an iterator
        self.i = 0; return self
    def __next__(self):
        if self.i >= self.n: raise StopIteration
        self.i += 1
        return self.i

for x in CountUp(3): print(x)     # 1 2 3
Generators — same protocol, far less code
def count_up(n):
    i = 0
    while i < n:
        i += 1
        yield i        # pauses here, resumes on the next next()

gen = count_up(3)
next(gen)          # 1 — body runs only up to the yield
next(gen)          # 2

# Generator expression — lazy, O(1) memory vs a list comprehension's O(n)
total = sum(x**2 for x in range(10**8))

under the hood▼
The GIL lets only one thread execute Python bytecode at a time. threading still gives real concurrency for I/O-bound work (the GIL is released during I/O waits) but no CPU parallelism. SOLID principles
OOD▼
SOLID is five principles for OOP that all push the same way: small units that depend on abstractions, so a change stays local instead of rippling. In interviews the two that come up most are Single Responsibility ("why is this class doing two jobs?") and Dependency Inversion ("inject that, don't hard-wire it"). Name the principle you're applying out loud — it signals intent.
The five, crisply S — Single Responsibility — one reason to change per class; if describing it needs "and", split it.
O — Open/Closed — open to extension, closed to modification; add a new subclass, don't edit working code.
L — Liskov Substitution — any subclass must be usable everywhere its base is, without surprising the caller (no strengthened preconditions, no weakened guarantees).
I — Interface Segregation — many small focused interfaces beat one fat one; don't force a class to implement methods it doesn't use.
D — Dependency Inversion — depend on abstractions (Protocol/ABC), not concrete classes; pass dependencies in rather than constructing them inside.
Dependency Inversion — inject the abstraction

patterns▼

    def notify(self, event):

    @abstractmethod
    def undo(self): ...          # pairing execute/undo enables undo-redo stacks
Exception design — EAFP, hierarchies & cleanup
error handling▼
Python favors EAFP ("easier to ask forgiveness than permission") — attempt the operation, catch the failure — over LBYL ("look before you leap"). EAFP is more Pythonic and avoids check-then-act race conditions. Design a small exception hierarchy per subsystem so callers can catch at exactly the granularity they care about.
EAFP vs LBYL
# LBYL — verbose and race-prone
if key in d:
    value = d[key]      # d could change between check and use (threaded)
else:
    value = default

# EAFP — Pythonic, atomic, one lookup
try:
    value = d[key]
except KeyError:
    value = default
Custom hierarchy — catch broad or narrow
class PaymentError(Exception): ...        # one base per subsystem
class CardDeclined(PaymentError): ...
class InsufficientFunds(PaymentError): ...

try:
    charge(card)
except InsufficientFunds:
    prompt_topup()       # handle the specific case
except PaymentError:
    log_and_retry()      # catch-all for the whole subsystem
Rules that matter in review
# Catch narrow, raise specific — never a bare `except:`
# (it swallows KeyboardInterrupt / SystemExit too)

# Preserve the original cause when re-raising:
try:
    charge(card)
except GatewayTimeout as err:
    raise PaymentError("charge failed") from err  # keeps the traceback chain

# Cleanup belongs in finally / a context manager, not copy-pasted
# into every except branch:
with open(path) as f:      # file closed even if the body raises
    process(f)

# Fail fast: validate inputs at the boundary and raise a clear,
# specific error — don't let bad data flow deep, then crash cryptically
Low-level design & code quality checklist
best practices▼
Before you call a design "done" Single responsibility — each class/function does one thing; if describing it needs "and", split it.
Composition over inheritance — inherit only for genuine is-a relationships; otherwise delegate to a member.
Encapsulate state — expose behavior, not raw mutable fields; validate at the boundary (@property), not scattered through the codebase.
Favor immutability — frozen dataclasses / tuples / namedtuples where state shouldn't change; fewer invariants to protect.
Name for intent, not implementation — get_active_users(), not get_data_where_flag_true().
Type hints on public interfaces — def f(x: int) -> list[str]: documents the contract and catches misuse early.
Depend on abstractions — accept a Protocol/ABC, not a concrete class, when a function just needs "something with .foo()".
Keep __init__ cheap — no I/O, no heavy computation; use a @classmethod factory for expensive construction.
Fail fast, fail clearly — validate inputs early and raise a specific exception with a useful message, not a generic one deep in the call stack.

Async/await fundamentals
foundation▼
async def defines a coroutine function — calling it returns a coroutine object, nothing runs yet. await suspends the current coroutine at an I/O point, handing control back to the event loop so other coroutines can run. This is single-threaded cooperative multitasking, not parallelism.
Coroutines vs. actually running them
async def fetch(url):
    ...                       # some awaited I/O inside
    return result

coro = fetch(url)             # just an object — nothing has run yet
result = await coro          # NOW it runs, to completion or exception

# Top-level entry point — starts the event loop
import asyncio
result = asyncio.run(fetch(url))
The #1 rule: never block the event loop
async def bad():
    time.sleep(2)             # BLOCKS the whole loop — nothing else
                            # can run for 2 seconds!

async def good():
    await asyncio.sleep(2)   # yields control back to the loop

# Same rule for CPU-heavy work — offload it to a thread/process:
async def cpu_heavy():
    loop = asyncio.get_running_loop()
    return await loop.run_in_executor(None, blocking_cpu_fn, data)
Concurrent requests — asyncio.gather / TaskGroup
fan-out▼
Fire off many independent API calls concurrently instead of awaiting them one at a time — the win comes from overlapping I/O wait time, not parallel CPU work.
gather — the classic fan-out
import asyncio, httpx

async def fetch(client, url):
    r = await client.get(url, timeout=5)
    r.raise_for_status()
    return r.json()

async def fetch_all(urls):
    async with httpx.AsyncClient() as client:
        tasks = [fetch(client, u) for u in urls]
        return await asyncio.gather(*tasks)      # runs all concurrently

# A single failure propagates by default; pass return_exceptions=True
# to get exceptions back as values instead of raising:
results = await asyncio.gather(*tasks, return_exceptions=True)
create_task & structured concurrency
async def fetch_all(urls):
    async with httpx.AsyncClient() as client:
        # create_task starts it eagerly — a bare coroutine only starts
        # running once you actually await it
        tasks = [asyncio.create_task(fetch(client, u)) for u in urls]
        return await asyncio.gather(*tasks)

# Python 3.11+ — cancels sibling tasks automatically if one errors
async def fetch_all_tg(urls):
    async with httpx.AsyncClient() as client, asyncio.TaskGroup() as tg:
        tasks = [tg.create_task(fetch(client, u)) for u in urls]
    return [t.result() for t in tasks]
Rate limiting & bounded concurrency
throttling▼
Firing thousands of requests at once just gets you rate-limited (or banned). Bound concurrency with a Semaphore, and respect the server's own signals — 429 status, Retry-After header.
Semaphore — cap concurrent in-flight requests
sem = asyncio.Semaphore(10)          # at most 10 requests in flight

async def fetch_bounded(client, url):
    async with sem:                  # acquire; released automatically
        r = await client.get(url, timeout=5)
        r.raise_for_status()
        return r.json()

results = await asyncio.gather(*(fetch_bounded(client, u) for u in urls))
Respecting 429 / Retry-After
async def fetch_with_backoff(client, url):
    r = await client.get(url, timeout=5)
    if r.status_code == 429:
        wait = float(r.headers.get("Retry-After", 1))
        await asyncio.sleep(wait)
        return await fetch_with_backoff(client, url)  # retry, simplified
    r.raise_for_status()
    return r.json()
Retries with exponential backoff
resilience▼
Only retry transient, idempotent-safe failures — timeouts, connection errors, 5xx, 429. Never blindly retry a plain 4xx (it'll fail identically) or a non-idempotent POST without an idempotency key. Add jitter so many clients don't retry in lockstep and stampede the server.
Exponential backoff with jitter
import random, asyncio

async def fetch_with_retry(client, url, max_attempts=5):
    for attempt in range(max_attempts):
        try:
            r = await client.get(url, timeout=5)
            if r.status_code >= 500 or r.status_code == 429:
                raise httpx.HTTPStatusError("retryable", request=r.request, response=r)
            r.raise_for_status()          # non-429 4xx -> raises, no retry
            return r.json()
        except (httpx.TimeoutException, httpx.HTTPStatusError):
            if attempt == max_attempts - 1:
                raise
            delay = (2 ** attempt) + random.uniform(0, 1)  # backoff + jitter
            await asyncio.sleep(delay)
Designing an API client wrapper
class design▼
Don't scatter requests/httpx calls throughout the codebase — wrap the API behind a small class. Callers get typed methods and a stable interface; you get one place to change auth, retries, and error handling.
A clean client wrapper
class APIError(Exception):
    def __init__(self, status_code, message):
        self.status_code = status_code
        super().__init__(message)

class UserClient:
    def __init__(self, base_url, token, client=None):
        self._base_url = base_url.rstrip("/")
        self._headers = {"Authorization": f"Bearer {token}"}
        self._client = client or httpx.AsyncClient()  # inject for testing

    async def get_user(self, user_id: int) -> dict:
        r = await self._client.get(
            f"{self._base_url}/users/{user_id}",
            headers=self._headers, timeout=5,
        )
        if r.status_code == 404:
            raise APIError(404, f"user {user_id} not found")
        r.raise_for_status()
        return r.json()

    async def __aenter__(self): return self
    async def __aexit__(self, *exc): await self._client.aclose()

async with UserClient(base_url, token) as client:
    user = await client.get_user(42)   # caller never touches httpx directly
Pagination patterns
pagination▼
Hide pagination behind a generator so callers just iterate — they shouldn't need to know whether the API uses page numbers, offsets, or opaque cursors.
Sync generator over pages
def iter_all_users(client, base_url):
    url = f"{base_url}/users"
    while url:
        r = client.get(url, timeout=5)
        r.raise_for_status()
        payload = r.json()
        yield from payload["results"]
        url = payload.get("next")     # cursor-style: server gives the next URL

for user in iter_all_users(client, base_url):
    process(user)                       # caller doesn't know/care about paging
Async generator — same idea, awaited
async def iter_all_users(client, base_url):
    url = f"{base_url}/users"
    while url:
        r = await client.get(url, timeout=5)
        r.raise_for_status()
        payload = r.json()
        for item in payload["results"]:
            yield item
        url = payload.get("next")

async for user in iter_all_users(client, base_url):
    process(user)
Timeouts & cancellation
resilience▼
Every network call needs both a per-request timeout (so one slow call can't hang forever) and often an overall deadline for a batch of work. asyncio.wait_for cancels the awaited coroutine if it overruns — handle CancelledError/TimeoutError where you can actually react, not deep inside a shared helper.
Per-call and overall timeouts
import asyncio

async def fetch_one(client, url):
    return await client.get(url, timeout=5)   # per-request timeout (httpx)

async def fetch_batch(client, urls):
    try:
        return await asyncio.wait_for(
            asyncio.gather(*(fetch_one(client, u) for u in urls)),
            timeout=20,                        # overall deadline for the whole batch
        )
    except asyncio.TimeoutError:
        return []                              # decide the fallback explicitly

# A cancelled coroutine gets asyncio.CancelledError raised at its
# current await point — clean up (locks, files) in a `finally`,
# and re-raise unless you deliberately want to swallow it

API integration checklist
best practices▼
Before you ship an API integration Always set a timeout on every call — a hung server should never hang your program.
Never call blocking I/O (requests, time.sleep, blocking DB drivers) inside async def — it stalls the whole event loop.
Bound concurrency with a Semaphore rather than firing every request at once.
Retry only transient, idempotent-safe failures — with backoff + jitter and a max-attempt cap.
Wrap the API behind a client class — one seam to change auth/retries/error handling, and to inject a fake in tests.
Model responses with real types (dataclass/pydantic), not raw dicts threaded through the whole codebase.
Close what you open — use async with / context managers for sessions and clients so connections don't leak.
Building the API yourself? Make writes idempotent, return meaningful status codes, and version the contract before you need to break it.
